128 void DNSNames::extract_resource_records (const DNSQueryBuffer* query_buffer, const int buffer_length) {
129  ns_msg message_handle; // Handle for the DNS response parsing structures
130  ns_rr resource_list; // List of resource records
131 
132  int resource_index; // Index of resource records in the resource list
133 
134  // Attempt to create a parser for the response from the DNS server. Not all resolver libraries require
135  // this call
136  if (ns_initparse (query_buffer->buf, buffer_length, &message_handle) < 0) {
137  throw DNSResolverException ("Failed to create a parser for the DNS response");
138  }
139 
140  // Walk the list of resource records, looking for those of the specified
141  // type. If we find a needed record, add it to the list of records to return
142  for (resource_index = 0; resource_index < ns_msg_count (message_handle, ns_s_an); resource_index++) {
143 
144  // Extract resource at the given index, and attempt to create a resource record
145  // for that resource
146  if (ns_parserr (&message_handle, ns_s_an, resource_index, &resource_list)) {
147  throw DNSResolverException ("Unable to extract the resource record from the listed index");
148  }
149 
150  // Do we have at least one record in the answer section?
151  if (ns_msg_count (message_handle, ns_s_an) < 1) {
152  throw DNSResolverException ("No records returned in the DNS answer section");
153  }
154 
155  if (ns_rr_type (resource_list) == convert_to_ns_type (cv_dns_query_type)) {
156  u_int tmp_resource_priority; // Temporary storage for the resourece priority
157  char* tmp_resource_name; // Temporary storage for the resource name
158 
159  //
160  // Extract the data, according to the type of record
161  //
162  switch (cv_dns_query_type) {
163  case DNSQueryType::A:
164  // Allocate storage for the resource name
165  tmp_resource_name = (char*) malloc (INET_ADDRSTRLEN);
166 
167  if (tmp_resource_name == nullptr) {
168  throw DNSResolverException ("Allocation of memory for resource data failed");
169  }
170

172  // and add it to the record list
173  inet_ntop (AF_INET, ns_rr_rdata (resource_list), tmp_resource_name, INET_ADDRSTRLEN);

180

183  }
184 
185  // Expand the resource name from data section of the resource list,

188 
189  break;
