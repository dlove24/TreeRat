259  // this call

261  throw DNSResolverException ("Failed to create a parser for the DNS response");
262  }

266  for (resource_index = 0; resource_index < ns_msg_count (message_handle, ns_s_an); resource_index++) {

276  throw DNSResolverException ("No records returned in the DNS answer section");
277  }
