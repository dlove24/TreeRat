20 static int auxresume (lua_State* L, lua_State* co, int narg) {
21  int status;
22 
23  if (!lua_checkstack (co, narg)) {
24  lua_pushliteral (L, "too many arguments to resume");
25  return -1; /* error flag */
26  }
27 
28  if (lua_status (co) == LUA_OK && lua_gettop (co) == 0) {
29  lua_pushliteral (L, "cannot resume dead coroutine");
30  return -1; /* error flag */
31  }
32 
33  lua_xmove (L, co, narg);
34  status = lua_resume (co, L, narg);
35 
36  if (status == LUA_OK || status == LUA_YIELD) {
37  int nres = lua_gettop (co);
38 
39  if (!lua_checkstack (L, nres + 1)) {
40  lua_pop (co, nres); /* remove results anyway */
41  lua_pushliteral (L, "too many results to resume");
42  return -1; /* error flag */
43  }
44 
45  lua_xmove (co, L, nres); /* move yielded values */
46  return nres;
47  }
48 
49  else {
50  lua_xmove (co, L, 1); /* move error message */
51  return -1; /* error flag */
52  }
53  }
